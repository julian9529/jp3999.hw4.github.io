  mutate(department = fct_reorder(department, days_since_prior_order)) %>%

#Barchart

instacartdf %>% 
  count(department) %>% 
  mutate(department = fct_reorder(department, n)) %>% 
  plot_ly(x = ~department, y = ~n, color = ~department, type = "bar", colors = "viridis")
